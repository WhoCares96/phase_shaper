84/* pointers to glist and array elements go through a "stub" which sticks
85around after the glist or array is freed. The stub itself is deleted when
86both the glist/array is gone and the refcount is zero, ensuring that no
87gpointers are pointing here. */
88
89#define GP_NONE 0 /* the stub points nowhere (has been cut off) */
90#define GP_GLIST 1 /* the stub points to a glist element */
91#define GP_ARRAY 2 /* ... or array */
92
93typedef struct _gstub
94{
95 union
96 {
97 struct _glist *gs_glist; /* glist we're in */
98 struct _array *gs_array; /* array we're in */
99 } gs_un;
100 int gs_which; /* GP_GLIST/GP_ARRAY */
101 int gs_refcount; /* number of gpointers pointing here */
102} t_gstub;
103
104typedef struct _gpointer /* pointer to a gobj in a glist */
105{
106 union
107 {
108 struct _scalar *gp_scalar; /* scalar we're in (if glist) */
109 union word *gp_w; /* raw data (if array) */
110 } gp_un;
111 int gp_valid; /* number which must match gpointee */
112 t_gstub *gp_stub; /* stub which points to glist/array */
113} t_gpointer;
